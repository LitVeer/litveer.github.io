冒泡排序

#include<stdio.h>

int main(){
    int a, b, c;
    scanf("%d %d %d",&a, &b, &c);
    int list[] = {a, b, c};
    for(int i=1; i <=3; i++){
        for(int j=i; i <=3; j++){
            if(list[j-1] < list[j]){
                list[j-1] ^= list[j] ^=list[j-1] ^= list[j];
            }
        }
    }
    printf("%d %d %d",list[0], list[1], list[2]);
    return 0;
}
